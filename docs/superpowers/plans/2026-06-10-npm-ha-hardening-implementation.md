### Task 1: DRBD task hardening — fix critical bug, add device wait, add sync-rate limit

**Files:**
- Modify: `roles/npm_ha/tasks/drbd.yml`
- Modify: `roles/npm_ha/templates/npm-ha.res.j2`
- Modify: `roles/npm_ha/defaults/main.yml`

This is the highest-priority fix: `drbd_init` is only registered when
`drbd_md_check.rc != 0`. On every re-run against the already-deployed
production nodes (where `drbd_md_check.rc == 0`), referencing
`drbd_init.changed` raises `AnsibleUndefinedVariable` and halts the whole
play. While in this file, also add a wait for the DRBD device nodes to
appear before formatting (they're attached asynchronously by the kernel
after `drbdadm primary --force all`), and add a sync-rate limit to the DRBD
resource template so the initial full sync doesn't saturate the replication
link.

Edit `roles/npm_ha/defaults/main.yml`. Find this block:

```yaml

drbd_resource_db: mib_npm_ha_drbd_db
```

Replace it with:

```yaml

drbd_resource_db: mib_npm_ha_drbd_db
```

- [ ] **Step 1: Rewrite `cluster.yml`**

Replace the entire content of `roles/npm_ha/tasks/cluster.yml` with:

```yaml
---
- name: Check if our cluster is already configured
  command: grep -q "{{ cluster_name }}" /etc/corosync/corosync.conf
  register: corosync_conf
  ignore_errors: true
  changed_when: false

- name: Remove dummy Debian corosync config if present
  file:
    path: /etc/corosync/corosync.conf
    state: absent
  when: corosync_conf.rc != 0

- name: Authenticate cluster nodes
  command: >
    pcs host auth {{ node1_hostname }} {{ node2_hostname }}
    -u hacluster -p {{ hacluster_password }}
  register: pcs_host_auth
  until: pcs_host_auth.rc == 0
  retries: 3
  delay: 5
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: corosync_conf.rc != 0
  no_log: true

- name: Setup the cluster
  command: >
    pcs cluster setup {{ cluster_name }}
    {{ node1_hostname }} addr={{ node1_ip }}
    {{ node2_hostname }} addr={{ node2_ip }}
    --force
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: corosync_conf.rc != 0

- name: Start and enable cluster services
  command: "{{ item }}"
  loop:
    - pcs cluster start --all
    - pcs cluster enable --all
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: corosync_conf.rc != 0

- name: Wait for both cluster nodes to be online
  command: pcs status nodes corosync
  register: corosync_nodes
  until: >-
    (node1_hostname + ' ' + node2_hostname) in corosync_nodes.stdout
    or (node2_hostname + ' ' + node1_hostname) in corosync_nodes.stdout
  retries: 24
  delay: 5
  changed_when: false
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: corosync_conf.rc != 0

- name: Set global cluster properties
  command: "{{ item }}"
  loop:
    - pcs property set stonith-enabled=false
    - pcs property set no-quorum-policy=ignore
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: corosync_conf.rc != 0
```

Changes vs. original: "Authenticate cluster nodes" gains `register`,
`until`, `retries: 3`, `delay: 5`, `no_log: true`; "Wait for cluster to
stabilise" (`pause`) is replaced by "Wait for both cluster nodes to be
online" (polling `pcs status nodes corosync`).

```yaml
  delegate_to: "{{ node1_hostname }}"

  command: pcs resource group add npm_group fs_app fs_db npm_vip npm_service
  run_once: true
  delegate_to: "{{ node1_hostname }}"

  command: "{{ item }}"
  loop:
    - pcs constraint colocation add promoted drbd_db-clone with promoted drbd_app-clone INFINITY
    - pcs constraint order promote drbd_app-clone then promote drbd_db-clone
    - pcs constraint colocation add npm_group with promoted drbd_db-clone INFINITY
    - pcs constraint order promote drbd_db-clone then start npm_group
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: group_check.rc != 0

- name: Update resource timeouts
  command: "{{ item }}"
  loop:
    - pcs resource update drbd_app op stop timeout={{ drbd_stop_timeout }}
    - pcs resource update drbd_db op stop timeout={{ drbd_stop_timeout }}
    - >
      pcs resource update npm_service
      op start timeout={{ npm_service_timeout }}
      op stop timeout={{ npm_service_timeout }}
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: group_check.rc != 0

- name: Wait for npm_group resources to start
  shell: pcs status resources | grep npm_service
  register: npm_service_status
  until: "'Started' in npm_service_status.stdout"
  retries: 24
  delay: 5
  changed_when: false
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: group_check.rc != 0

- name: Ensure application subdirectories exist on DRBD mounts

    path: "{{ item }}"
    state: directory
    owner: root
    group: root
    mode: '0755'

    - "{{ npm_mount_app }}/letsencrypt"
    - "{{ npm_mount_db }}/db/mysql"
  run_once: true
  delegate_to: "{{ node1_hostname }}"
  when: group_check.rc != 0

- name: Final cleanup — reset any Pacemaker failure states

  run_once: true
  delegate_to: "{{ node1_hostname }}"
  changed_when: false
```
